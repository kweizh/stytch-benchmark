```python
import requests
import sys
from requests.auth import HTTPBasicAuth

def setup_jit(organization_id):
    project_id = "project-test-00000000-0000-0000-0000-000000000000"
    secret = "secret-test-11111111-1111-1111-1111-111111111111"
    url = f"https://test.stytch.com/v1/b2b/organizations/{organization_id}"
    
    payload = {
        "oauth_tenant_jit_provisioning": "RESTRICTED",
        "allowed_oauth_tenants": [
            {"provider": "slack", "tenant_id": "SLACK-123"},
            {"provider": "github", "tenant_id": "12345"}
        ]
    }
    
    response = requests.put(
        url,
        json=payload,
        auth=HTTPBasicAuth(project_id, secret)
    )
    
    print(response.status_code)

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python setup_jit.py <organization_id>")
        sys.exit(1)
    
    org_id = sys.argv[1]
    setup_jit(org_id)
```   
